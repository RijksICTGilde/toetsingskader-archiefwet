{{- /* Render tekst met {ref:N} → interactieve referentie-tooltips. N = positie (1-based) in .Params.referenties lijst. Bij hover/focus verschijnt een tooltip met de referentietekst en eventuele bronlink. Gebruik {ref:N} i.p.v. [ref:N] om conflict met markdown link-syntax te voorkomen. Gebruik: {{ partial "reftext.html" (dict "tekst" .tekst "page" $page) }} */ -}} {{- $tekst := .tekst | markdownify -}} {{- $base := "x" | relURL -}} {{- $base = strings.TrimSuffix "x" $base -}} {{- if ne $base "/" -}} {{- $tekst = replace $tekst "href=\"/" (printf "href=\"%s" $base) -}} {{- end -}} {{- $page := .page -}} {{- range $i, $ref := $page.Params.referenties -}} {{- $num := add $i 1 -}} {{- $search := printf "{ref:%d}" $num -}} {{- $label := printf "Referentie %d: %s" $num $ref.tekst | htmlEscape -}} {{- $textEsc := $ref.tekst | htmlEscape -}} {{- $linkHTML := "" -}} {{- with $ref.url -}} {{- $linkHTML = printf "Bekijk bron ↗" . -}} {{- end -}} {{- $icon := "" -}} {{- $replace := printf "%s%s" $label $icon $textEsc $linkHTML -}} {{- $tekst = replace $tekst $search ($replace | safeHTML) -}} {{- end -}} {{- $tekst | safeHTML -}}
