{{- /* Render tekst met {ref:N} → interactieve referentie-tooltips. N = positie (1-based) in .Params.referenties lijst. Emit rijke markup met zowel cijfer, icoon als voorafgaand woord; CSS op [data-ref-variant="1..4"] op bepaalt welke variant zichtbaar is. Zonder data-ref-variant = huidige styling (icoon + hover). Gebruik: {{ partial "reftext.html" (dict "tekst" .tekst "page" $page) }} */ -}} {{- $tekst := .tekst | markdownify -}} {{- $base := "x" | relURL -}} {{- $base = strings.TrimSuffix "x" $base -}} {{- if ne $base "/" -}} {{- $tekst = replace $tekst "href=\"/" (printf "href=\"%s" $base) -}} {{- end -}} {{- $page := .page -}} {{- range $i, $ref := $page.Params.referenties -}} {{- $num := add $i 1 -}} {{- $hasUrl := "false" -}} {{- if $ref.url -}}{{- $hasUrl = "true" -}}{{- end -}} {{- $label := printf "Referentie %d: %s" $num $ref.tekst | htmlEscape -}} {{- $textEsc := $ref.tekst | htmlEscape -}} {{- $linkHTML := "" -}} {{- with $ref.url -}} {{- $linkHTML = printf "Bekijk bron ↗" . -}} {{- end -}} {{- $icon := "" -}} {{- $tooltip := printf "%s%s" $textEsc $linkHTML -}} {{- $anchor := printf "%d" $num $label $num -}} {{- $button := printf "" $label $icon -}} {{- $wrapperOpen := printf "" $num $hasUrl -}} {{- $replacement := printf "%s${1}%s%s%s${2}" $wrapperOpen $anchor $button $tooltip -}} {{- $pattern := printf "([^\\s<>]*)(\\s*)\\{ref:%d\\}" $num -}} {{- $tekst = replaceRE $pattern $replacement $tekst -}} {{- end -}} {{- $tekst | safeHTML -}}
