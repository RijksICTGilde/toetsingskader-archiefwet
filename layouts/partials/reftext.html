{{- /* Render tekst met {ref:N} → onderstippelde underline op het direct voorafgaande woord. Hover/focus toont een tooltip met referenties[N-1].tekst en (indien aanwezig) een Bekijk-bron-link. Gebruik: {{ partial "reftext.html" (dict "tekst" .tekst "page" $page) }} */ -}} {{- $tekst := .tekst | markdownify -}} {{- $base := "x" | relURL -}} {{- $base = strings.TrimSuffix "x" $base -}} {{- if ne $base "/" -}} {{- $tekst = replace $tekst "href=\"/" (printf "href=\"%s" $base) -}} {{- end -}} {{- $page := .page -}} {{- range $i, $ref := $page.Params.referenties -}} {{- $num := add $i 1 -}} {{- $textEsc := $ref.tekst | htmlEscape -}} {{- $linkHTML := "" -}} {{- with $ref.url -}} {{- $linkHTML = printf "Bekijk bron ↗" . -}} {{- end -}} {{- $tooltip := printf "%s%s" $textEsc $linkHTML -}} {{- $pattern := printf "([^\\s<>]*)(\\s*)\\{ref:%d\\}" $num -}} {{- $replace := printf "${1}%s${2}" $tooltip -}} {{- $tekst = replaceRE $pattern $replace $tekst -}} {{- end -}} {{- $tekst | safeHTML -}}
