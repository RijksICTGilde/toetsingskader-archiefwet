{{- /* Render tekst met {ref:N} → term met onderstippelde underline en hover/ focus-tooltip met de bijbehorende referentie-uitleg. N = positie (1-based) in .Params.referenties lijst. Gebruik: {{ partial "reftext.html" (dict "tekst" .tekst "page" $page) }} */ -}} {{- $tekst := .tekst | markdownify -}} {{- $base := "x" | relURL -}} {{- $base = strings.TrimSuffix "x" $base -}} {{- if ne $base "/" -}} {{- $tekst = replace $tekst "href=\"/" (printf "href=\"%s" $base) -}} {{- end -}} {{- $page := .page -}} {{- range $i, $ref := $page.Params.referenties -}} {{- $num := add $i 1 -}} {{- $textEsc := $ref.tekst | htmlEscape -}} {{- $linkHTML := "" -}} {{- with $ref.url -}} {{- $linkHTML = printf "Bekijk bron ↗" . -}} {{- end -}} {{- $tooltip := printf "%s%s" $textEsc $linkHTML -}} {{- $replacement := printf "${1}%s${2}" $tooltip -}} {{- $pattern := printf "([^\\s<>]*)(\\s*)\\{ref:%d\\}" $num -}} {{- $tekst = replaceRE $pattern $replacement $tekst -}} {{- end -}} {{- $tekst | safeHTML -}}
